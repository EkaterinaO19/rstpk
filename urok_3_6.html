from mcpi.minecraft import Minecraft mc = Minecraft.create() # Получаем данные о Стиве pos = mc.player.getPos() hp = int(input("Сколько у тебя HP? ")) blok = mc.getBlock(pos.x, pos.y - 1, pos.z) # Умный анализ ситуации if blok == 10 or blok == 51: # лава/огонь mc.postToChat("🔥 ОПАСНОСТЬ! Телепорт!") mc.player.setPos(pos.x, pos.y + 20, pos.z) elif hp <= 5 and blok != 9: # мало HP, не в воде mc.postToChat("❤️ HP критично! Иди домой!") mc.player.setPos(0, 64, 0) elif blok == 56: # алмазная руда mc.postToChat("💎 Алмазная руда! Копай!") else: mc.postToChat(f"✅ Всё ок. Блок: {blok}, HP: {hp}")

Что умеет getBlock() в сочетании с if?
Только проверять один тип блока
Определять любой блок и реагировать по-разному
Только камень и траву
Ничего — нельзя сравнивать числа
Зачем blok != 9 в условии hp <= 5 and blok != 9?
Чтобы не телепортировать из воды дважды
Чтобы проверить не стоит ли игрок в воде — вода лечит
Это лишняя проверка
Чтобы число 9 не вызывало ошибку
Каков правильный порядок проверок в умном помощнике?
Сначала else, потом if
Самые важные/срочные условия — первыми в if
Порядок не важен
Сначала else, потом elif, потом if
